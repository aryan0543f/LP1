---------------------FIRST IN FIRST OUT------------------



package synchronization;

import java.util.*;

public class FIFO {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter number of pages: ");
        int n = sc.nextInt();
        int pages[] = new int[n];
        System.out.println("Enter reference string:");
        for (int i = 0; i < n; i++) 
            pages[i] = sc.nextInt();

        System.out.print("Enter frame capacity: ");
        int cap = sc.nextInt();
        int frame[] = new int[cap];
        Arrays.fill(frame, -1);

        int index = 0, hit = 0, fault = 0;

        System.out.println("\nPage\tFrames\t\tStatus");
        System.out.println("----\t------\t\t------");
        
        for (int i = 0; i < n; i++) {
            boolean found = false;

            // Check for hit
            for (int j = 0; j < cap; j++) {
                if (frame[j] == pages[i]) {
                    found = true;
                    hit++;
                    break;
                }
            }

            // If not found => page fault
            if (!found) {
                frame[index] = pages[i];
                index = (index + 1) % cap;  // Circular replacement
                fault++;
            }

            // Print current status
            System.out.print(pages[i] + "\t");
            for (int j = 0; j < cap; j++) {
                if (frame[j] == -1)
                    System.out.print("- ");
                else
                    System.out.print(frame[j] + " ");
            }
            System.out.println("\t\t" + (found ? "Hit" : "Fault"));
        }

        System.out.println("\n=== RESULTS ===");
        System.out.println("Total Hits: " + hit);
        System.out.println("Total Faults: " + fault);
        System.out.printf("Hit Ratio: %.2f%%\n", (hit * 100.0 / n));
        System.out.printf("Fault Ratio: %.2f%%\n", (fault * 100.0 / n));

        sc.close();
    }
}

import java.util.*;

public class OptimalPageReplacementAlgorithm {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        
        System.out.print("Enter number of pages: ");
        int n = sc.nextInt();
        int pages[] = new int[n];
        System.out.println("Enter reference string:");
        for (int i = 0; i < n; i++) 
            pages[i] = sc.nextInt();

        System.out.print("Enter number of frames: ");
        int f = sc.nextInt();
        int frame[] = new int[f];
        Arrays.fill(frame, -1);
        
        int hit = 0, fault = 0;

        System.out.println("\nPage\tFrames\t\tStatus");
        System.out.println("----\t------\t\t------");
        
        for (int i = 0; i < n; i++) {
            int p = pages[i];
            int pos = -1;
            
            // Check if page already exists
            for (int j = 0; j < f; j++) 
                if (frame[j] == p) 
                    pos = j;

            if (pos == -1) { // Page fault
                fault++;
                int replaceIndex = -1;
                int farthest = -1;
                
                // Find empty frame first
                for (int j = 0; j < f; j++) {
                    if (frame[j] == -1) { 
                        replaceIndex = j; 
                        break; 
                    }
                }
                
                // If no empty frame, find optimal replacement
                if (replaceIndex == -1) {
                    for (int j = 0; j < f; j++) {
                        int k;
                        // Find when this frame's page will be used next
                        for (k = i + 1; k < n; k++) 
                            if (frame[j] == pages[k]) 
                                break;
                        
                        // If this page is used farthest in future (or never)
                        if (k > farthest) { 
                            farthest = k; 
                            replaceIndex = j; 
                        }
                    }
                }
                
                frame[replaceIndex] = p;
            } else {
                hit++;
            }

            // Print current status
            System.out.print(p + "\t");
            for (int j = 0; j < f; j++) {
                if (frame[j] == -1)
                    System.out.print("- ");
                else
                    System.out.print(frame[j] + " ");
            }
            System.out.println("\t\t" + (pos == -1 ? "Fault" : "Hit"));
        }
        
        System.out.println("\n=== RESULTS ===");
        System.out.println("Total Hits: " + hit);
        System.out.println("Total Faults: " + fault);
        System.out.printf("Hit Ratio: %.2f%%\n", (hit * 100.0 / n));
        System.out.printf("Fault Ratio: %.2f%%\n", (fault * 100.0 / n));
        
        sc.close();
    }
}

import java.util.*;

public class LRUPageReplacement {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter number of pages: ");
        int n = sc.nextInt();
        int pages[] = new int[n];
        System.out.println("Enter reference string:");
        for (int i = 0; i < n; i++) 
            pages[i] = sc.nextInt();

        System.out.print("Enter frame capacity: ");
        int f = sc.nextInt();
        int frame[] = new int[f];
        Arrays.fill(frame, -1);
        
        int hit = 0, fault = 0;

        System.out.println("\nPage\tFrames\t\tStatus");
        System.out.println("----\t------\t\t------");
        
        // Track recent usage order
        ArrayList<Integer> recent = new ArrayList<>();

        for (int i = 0; i < n; i++) {
            int p = pages[i];
            boolean found = false;

            // Check for hit
            for (int j = 0; j < f; j++) {
                if (frame[j] == p) { 
                    found = true; 
                    hit++; 
                    break; 
                }
            }

            // If not found => page fault
            if (!found) {
                fault++;
                int replaceIndex = -1;
                
                // Find empty frame first
                for (int j = 0; j < f; j++) {
                    if (frame[j] == -1) { 
                        replaceIndex = j; 
                        break; 
                    }
                }

                // If no empty frame â†’ remove least recently used
                if (replaceIndex == -1) {
                    int lruPage = recent.get(0);  // First in list = LRU
                    for (int j = 0; j < f; j++) {
                        if (frame[j] == lruPage) { 
                            replaceIndex = j; 
                            break; 
                        }
                    }
                    recent.remove(0);  // Remove LRU from tracking
                }
                
                frame[replaceIndex] = p;
            }

            // Update recent usage (most recent at end)
            recent.remove((Integer)p);  // Remove if already exists
            recent.add(p);              // Add to end (most recent)

            // Print current status
            System.out.print(p + "\t");
            for (int j = 0; j < f; j++) {
                if (frame[j] == -1)
                    System.out.print("- ");
                else
                    System.out.print(frame[j] + " ");
            }
            System.out.println("\t\t" + (found ? "Hit" : "Fault"));
        }

        System.out.println("\n=== RESULTS ===");
        System.out.println("Total Hits: " + hit);
        System.out.println("Total Faults: " + fault);
        System.out.printf("Hit Ratio: %.2f%%\n", (hit * 100.0 / n));
        System.out.printf("Fault Ratio: %.2f%%\n", (fault * 100.0 / n));
        
        sc.close();
    }
}

What are Page Replacement Algorithms?
Strategies used by operating systems to decide which page to remove from memory
when a new page needs to be loaded but memory is full (page fault occurs).

- Page Fault: When requested page is not in memory

- Frame: Physical memory slot that holds a page
- Reference String: Sequence of page requests from processes

ðŸ“‹ FIFO (First In First Out):
- Replaces the OLDEST page in memory

- Suffers from Belady's Anomaly (more frames â‰  fewer faults)
- Time Complexity: O(1) for replacement decision
- Not optimal but fair and predictable

ðŸŽ¯ OPTIMAL Page Replacement:
- Replaces page that will be used FARTHEST in future

- Provides MINIMUM possible page faults

- Replaces page that hasn't been used for LONGEST time

- Good approximation of optimal algorithm
- Requires tracking of page usage history
- More complex but better performance than FIFO
